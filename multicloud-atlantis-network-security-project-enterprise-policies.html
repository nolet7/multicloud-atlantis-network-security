1. Project Overview

This project builds a controlled multi-cloud network foundation using Terraform and Atlantis. Atlantis is the execution engine. Engineers do not run terraform apply from personal laptops. Instead, all infrastructure changes are proposed through GitHub pull requests, planned by Atlantis, reviewed by the platform/security team, and applied only after approval.

AWS

Create VPC, public/private subnets, internet gateway, route table, and security group.

Azure

Create resource group, virtual network, subnet, and network security group with inbound rules.

GCP

Create custom VPC, subnet, and firewall rules for SSH/HTTP from approved CIDRs.

Atlantis

Runs plan/apply remotely from PR comments and posts results back into GitHub.

Goal: A multi-cloud network/security baseline where every change is peer-reviewed, traceable, policy-checked, and applied centrally.

3. Atlantis Workflow

The workflow is intentionally PR-driven:

  1. Engineer creates a feature branch.
  2. Engineer edits Terraform code for AWS, Azure, or GCP.
  3. Engineer opens a pull request.
  4. Atlantis automatically runs terraform plan for changed project directories.
  5. Reviewer checks plan output and security scan results.
  6. Authorized engineer comments atlantis apply -p project-name.
  7. Atlantis applies the reviewed plan centrally.
Important: Developers should not run terraform apply locally. Local Terraform can be used only for formatting or static validation if your company allows it.

8. Deploy GCP VM for Atlantis

This section deploys a Google Compute Engine VM that runs Atlantis as a Docker container. The goal is still no local Terraform apply: local commands only bootstrap the Atlantis runner VM. After that, all AWS, Azure, and GCP network/security changes happen through pull requests and Atlantis.

Why GCP VM? Atlantis can run on Google Compute Engine. The official Atlantis deployment documentation lists Google Compute Engine as a supported deployment option and the official Atlantis image is available at ghcr.io/runatlantis/atlantis. This guide uses a simple VM bootstrap pattern for a lab and can later be upgraded to a managed instance group or GKE.

8.6 Create the Atlantis VM startup script

The startup script installs Docker, reads secrets from Secret Manager, writes a locked-down environment file, and starts Atlantis.

mkdir -p atlantis-vm
cat > atlantis-vm/startup-atlantis.sh <<'EOF'
#!/usr/bin/env bash
set -euo pipefail

apt-get update -y
apt-get install -y ca-certificates curl gnupg lsb-release google-cloud-cli docker.io
systemctl enable docker
systemctl start docker

PROJECT_ID=$(curl -s -H "Metadata-Flavor: Google" \
  http://metadata.google.internal/computeMetadata/v1/project/project-id)

read_secret() {
  local name="$1"
  gcloud secrets versions access latest --secret="$name" --project="$PROJECT_ID"
}

mkdir -p /opt/atlantis
cd /opt/atlantis

cat > /opt/atlantis/repos.yaml <<'REPOS'
repos:
  - id: /.*/
    branch: /main/
    apply_requirements: [approved, mergeable, undiverged]
    allowed_overrides: [workflow, apply_requirements, autoplan]
    allow_custom_workflows: false
    delete_source_branch_on_merge: true
REPOS

cat > /opt/atlantis/atlantis.env <<ENV
ATLANTIS_GH_USER=$(read_secret atlantis-gh-user)
ATLANTIS_GH_TOKEN=$(read_secret atlantis-gh-token)
ATLANTIS_GH_WEBHOOK_SECRET=$(read_secret atlantis-gh-webhook-secret)
ATLANTIS_REPO_ALLOWLIST=$(read_secret atlantis-repo-allowlist)
ATLANTIS_ATLANTIS_URL=$(read_secret atlantis-url)
AWS_ACCESS_KEY_ID=$(read_secret aws-access-key-id)
AWS_SECRET_ACCESS_KEY=$(read_secret aws-secret-access-key)
AWS_DEFAULT_REGION=$(read_secret aws-default-region)
ARM_CLIENT_ID=$(read_secret arm-client-id)
ARM_CLIENT_SECRET=$(read_secret arm-client-secret)
ARM_SUBSCRIPTION_ID=$(read_secret arm-subscription-id)
ARM_TENANT_ID=$(read_secret arm-tenant-id)
GOOGLE_PROJECT=$(read_secret google-project)
GOOGLE_REGION=$(read_secret google-region)
ENV

chmod 600 /opt/atlantis/atlantis.env

cat > /etc/systemd/system/atlantis.service <<'UNIT'
[Unit]
Description=Atlantis Terraform PR Automation
After=docker.service
Requires=docker.service

[Service]
Restart=always
EnvironmentFile=/opt/atlantis/atlantis.env
ExecStartPre=-/usr/bin/docker rm -f atlantis
ExecStart=/usr/bin/docker run --name atlantis \
  --env-file /opt/atlantis/atlantis.env \
  -v /opt/atlantis/repos.yaml:/etc/atlantis/repos.yaml:ro \
  -v atlantis-data:/home/atlantis \
  -p 4141:4141 \
  ghcr.io/runatlantis/atlantis:latest \
  server \
  --gh-user=$ATLANTIS_GH_USER \
  --gh-token=$ATLANTIS_GH_TOKEN \
  --gh-webhook-secret=$ATLANTIS_GH_WEBHOOK_SECRET \
  --repo-allowlist=$ATLANTIS_REPO_ALLOWLIST \
  --atlantis-url=$ATLANTIS_ATLANTIS_URL \
  --repo-config=/etc/atlantis/repos.yaml \
  --log-level=info
ExecStop=/usr/bin/docker stop atlantis

[Install]
WantedBy=multi-user.target
UNIT

systemctl daemon-reload
systemctl enable atlantis
systemctl start atlantis
EOF

mergeableThe PR must have no merge conflicts and must satisfy branch rules.Prevents applying code that cannot cleanly merge into main.

13.4 Example OPA/Rego policy: block open admin ports

mkdir -p policies/opa
cat > policies/opa/network_security.rego <<'EOF'
package terraform.network_security

deny[msg] {
  resource := input.resource_changes[_]
  resource.type == "aws_security_group_rule"
  resource.change.after.type == "ingress"
  resource.change.after.from_port <= 22
  resource.change.after.to_port >= 22
  resource.change.after.cidr_blocks[_] == "0.0.0.0/0"
  msg := "AWS security group must not expose SSH port 22 to 0.0.0.0/0"
}

deny[msg] {
  resource := input.resource_changes[_]
  resource.type == "azurerm_network_security_rule"
  resource.change.after.direction == "Inbound"
  resource.change.after.destination_port_range == "22"
  resource.change.after.source_address_prefix == "*"
  msg := "Azure NSG must not expose SSH port 22 to all sources"
}

deny[msg] {
  resource := input.resource_changes[_]
  resource.type == "google_compute_firewall"
  resource.change.after.source_ranges[_] == "0.0.0.0/0"
  allow := resource.change.after.allow[_]
  allow.ports[_] == "22"
  msg := "GCP firewall must not expose SSH port 22 to 0.0.0.0/0"
}
EOF

Explanation: This policy checks Terraform plan JSON before apply. It blocks public SSH exposure across AWS, Azure, and GCP.

13.5 Example OPA/Rego policy: require enterprise tags/labels

cat > policies/opa/required_metadata.rego <<'EOF'
package terraform.required_metadata

required_tags := {"Owner", "Environment", "ManagedBy"}

deny[msg] {
  resource := input.resource_changes[_]
  startswith(resource.type, "aws_")
  missing := required_tags - {tag | resource.change.after.tags[tag]}
  count(missing) > 0
  msg := sprintf("AWS resource %s is missing required tags: %v", [resource.address, missing])
}

deny[msg] {
  resource := input.resource_changes[_]
  startswith(resource.type, "azurerm_")
  missing := required_tags - {tag | resource.change.after.tags[tag]}
  count(missing) > 0

required_labels := {"owner", "environment", "managed_by"}

deny[msg] {
  resource := input.resource_changes[_]

EOF

Explanation: This policy enforces cost ownership, environment classification, and audit traceability. AWS/Azure use tags; GCP commonly uses labels.

18. Runbook

First-Time Build

  1. Generate repository files using the bootstrap commands.
  2. Replace placeholder org/repo values in repos.yaml and docker-compose.atlantis.yml.
  3. Configure cloud automation credentials securely.
  4. Start Atlantis.
  5. Create GitHub webhook to https://atlantis.yourdomain.com/events.
  6. Open PR with Terraform changes.
  7. Review Atlantis plan.
  8. Approve and apply through PR comment.

Production Hardening Checklist

  • Use TLS for Atlantis.
  • Place Atlantis behind private ingress or authenticated reverse proxy where possible.
  • Use workload identity or short-lived credentials instead of static secrets.
  • Separate AWS/Azure/GCP credentials by environment.
  • Enable GitHub branch protection and required CODEOWNERS approval.
  • Use remote Terraform backend with locking.
  • Stream Atlantis logs to central observability platform.
  • Backup Terraform state and enable object versioning.
Final result: Engineers only submit pull requests. Atlantis plans, reviewers approve, and Atlantis applies. This gives your organization a repeatable, auditable, no-local-apply multi-cloud network automation workflow.

Policy: Deny Public SSH

Purpose: Blocks AWS security groups, Azure NSGs, and GCP firewall rules that expose SSH to the internet. This protects the project from accidental administrative exposure.

# policies/deny-public-ssh.rego
package terraform.security

deny[msg] {
  resource := input.resource_changes[_]
  resource.type == "aws_security_group"
  ingress := resource.change.after.ingress[_]
  ingress.from_port == 22
  ingress.to_port == 22
  ingress.cidr_blocks[_] == "0.0.0.0/0"
  msg := sprintf("AWS security group %s exposes SSH to the internet", [resource.address])
}

deny[msg] {
  resource := input.resource_changes[_]
  resource.type == "azurerm_network_security_group"
  rule := resource.change.after.security_rule[_]
  rule.destination_port_range == "22"
  rule.source_address_prefix == "*"
  msg := sprintf("Azure NSG %s exposes SSH to the internet", [resource.address])
}

deny[msg] {
  resource := input.resource_changes[_]
  resource.type == "google_compute_firewall"
  rule := resource.change.after
  rule.allow[_].ports[_] == "22"
  rule.source_ranges[_] == "0.0.0.0/0"
  msg := sprintf("GCP firewall %s exposes SSH to the internet", [resource.address])
}

How the Full Deployment Works

  1. Engineer updates envs/dev/terraform.tfvars, envs/stage/terraform.tfvars, envs/prod/terraform.tfvars, or a shared module.
  2. Engineer opens a GitHub pull request.
  3. Atlantis detects the changed environment folder and runs a plan.
  4. OPA policies check for public SSH, missing tags, invalid environments, broad production CIDRs, and hardcoded secrets.
  5. Reviewers approve the PR.
  6. Authorized engineer comments atlantis apply -p multicloud-dev, multicloud-stage, or multicloud-prod.
  7. Atlantis applies the approved plan from the GCP VM using secrets from Secret Manager and cloud identities.

package main

admin_ports := {22, 3389, 5985, 5986, 6443, 5432, 3306, 6379, 9200, 9300}
valid_envs := {"dev", "stage", "prod"}
approved_regions := {"us-east-1", "us-west-2", "eastus", "centralus", "us-central1", "us-east1"}
required_tags := {"Project", "Environment", "Owner", "CostCenter", "ManagedBy"}

deny[msg] {
  rc := input.resource_changes[_]
  rc.type == "aws_security_group_rule"
  rc.change.after.type == "ingress"
  rc.change.after.cidr_blocks[_] == "0.0.0.0/0"
  port := admin_ports[_]
  rc.change.after.from_port <= port
  rc.change.after.to_port >= port
  msg := sprintf("Admin port %v cannot be open to the internet: %s", [port, rc.address])
}

deny[msg] {
  rc := input.resource_changes[_]
  rc.type == "google_compute_firewall"
  rc.change.after.source_ranges[_] == "0.0.0.0/0"
  allow := rc.change.after.allow[_]
  allow.ports[_] == "22"
  msg := sprintf("Public SSH is not allowed in GCP firewall: %s", [rc.address])
}

deny[msg] {
  rc := input.resource_changes[_]
  startswith(rc.type, "aws_")
  tags := object.get(rc.change.after, "tags", {})
  required := required_tags[_]
  not tags[required]
  msg := sprintf("Missing required AWS tag %s on %s", [required, rc.address])
}

deny[msg] {
  rc := input.resource_changes[_]
  startswith(rc.type, "azurerm_")
  tags := object.get(rc.change.after, "tags", {})
  required := required_tags[_]
  not tags[required]
  msg := sprintf("Missing required Azure tag %s on %s", [required, rc.address])
}

deny[msg] {
  rc := input.resource_changes[_]
  tags := object.get(rc.change.after, "tags", {})
  env := tags["Environment"]
  env != ""
  not valid_envs[env]
  msg := sprintf("Invalid Environment tag %s on %s", [env, rc.address])
}

deny[msg] {
  rc := input.resource_changes[_]
  rc.change.actions[_] == "delete"
  tags := object.get(rc.change.before, "tags", {})
  tags["Environment"] == "prod"
  msg := sprintf("Production destroy is blocked: %s", [rc.address])
}

deny[msg] {
  rc := input.resource_changes[_]
  region := object.get(rc.change.after, "region", "")
  region != ""
  not approved_regions[region]
  msg := sprintf("Region %s is not approved for %s", [region, rc.address])
}
